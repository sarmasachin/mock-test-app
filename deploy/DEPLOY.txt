Fresh server cleanup (run once on VPS as root)
----------------------------------------------
  rm -f /etc/nginx/sites-enabled/*
  rm -f /etc/nginx/sites-available/livehospital*
  rm -f /etc/nginx/sites-available/indiaapk*
  rm -f /etc/nginx/sites-available/admin-admin.livehospital*
  grep -r "livehospital\|indiaapk" /etc/nginx/ 2>/dev/null

Install nginx site
------------------
  cp deploy/nginx-site.govmocktest.conf.example /etc/nginx/sites-available/admin-admin.govmocktest.com
  ln -sf /etc/nginx/sites-available/admin-admin.govmocktest.com /etc/nginx/sites-enabled/
  nginx -t && systemctl reload nginx
  certbot --nginx -d admin-admin.govmocktest.com

Deploy admin static from Windows
--------------------------------
  cd admin-web
  echo VITE_API_BASE_URL=https://admin-admin.govmocktest.com/v1 > .env.production
  npm install && npm run build
  ..\deploy\deploy-admin.ps1 -ServerUser root -ServerHost YOUR_VPS_IP

Redeploy after git push
-----------------------
  cd /var/www/mocktestapp && git pull origin main
  cd server && npm install && pm2 restart mocktest-api
  cd ../admin-web && npm install && npm run build
  cp -r dist/* /var/www/admin-admin.govmocktest.com/site/admin/
  systemctl reload nginx
